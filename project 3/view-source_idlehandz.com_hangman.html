<script type="text/javascript">
$(document).ready(function(){
$("#stand").hide();
$("#rope").hide();
$("#head").hide();
$("#neck").hide();
$("#torso").hide();
$("#leftLeg").hide();
$("#rightLeg").hide();
createButtons();
$("#A").before("<br/>");
createLetters();
$("#A").before("<p>Misses: " + missCount + "/7</p><br/>");
$("button").click(function() {
var letterText = $(this).text().toLowerCase();
if (!gameOver)
guess(letterText);
$("p").text("Misses: " + missCount + "/7");
$(this).css("visibility", "hidden");
})
$("#startOver").click(function() {
location.reload();
});
});

var missCount = 0;
var hitCount = 0;
var gameOver = false;
const words = [
"test",
"solution",
"computer",
"science",
"javascript",
"react",
"rallante",
"northridge",
"supercalifragilisticexpialidocious",
"jquery",
"fritz",
"bubblegum",
"pudding",
"randomness",
"antidisestablishmentarianism",
"instansiate",
"exponentiate",
"indubitably",
"dastardly",
"negation",
"extenuating",
"confounded"
];
const randomIndex = Math.ceil(Math.random() * 100) % (words.length + 1);

const randomWord = words[randomIndex];
console.log("Word selected: " + randomWord);

const solution = randomWord.split('');
// var guessed = solution.map(() => ' ');

const range = (count, gen = (i => i)) => {
const array = [];
let current = 0;

while (current < count) {
array.push(gen(current));
current += 1;
}

return array;
}; // End range function
const alphas = range(
26,
i => String.fromCharCode(i + 'A'.charCodeAt(0))
);
function makeLetter(letter) {
return ("<letter class='" + letter + "'>&nbsp</letter>");
}
function createButtons(){
for (var i = alphas.length-1; i >=0; i--) {
$("#hangedMan").after("<button id='" + alphas[i] + "'>" + alphas[i] + "</button>");
//console.log(alphas[i] + " button created.");
}
console.log("Buttons created.");
}
function createLetters(){
for (var i = randomWord.length-1; i >= 0; i--)
{
$("#hangedMan").after(makeLetter(randomWord[i]));
}
console.log("Letters created.");
}

function guess(letterText) {
var index = randomWord.indexOf(letterText);
if (index >= 0){
$("letter."+letterText).text(""+letterText);
console.log(letterText + " correctly guessed!");
// console.log($("letter." + letterText).text() + "replaced.");
var occurences = randomWord.split(''+letterText).length-1;
for (var i = 0; i<occurences; i++)
{
hitCount++;
}
}
else {
console.log(letterText + " incorrectly guessed. Index: " + index);
missCount++;
}
if (hitCount == solution.length){
alert("Congrats! You Won!\nWord: " + randomWord);
gameOver = true;
}
if (missCount > 0){
$("#stand").show();
console.log("Stand revealed.");
}
if (missCount > 1){
$("#rope").show();
console.log("Rope revealed.");
}
if (missCount > 2){
$("#head").show();
console.log("Head revealed.");
}
if (missCount > 3){
$("#neck").show();
console.log("Neck revealed.");
}
if (missCount > 4){
$("#torso").show();
console.log("Torso revealed.");
}
if (missCount > 5){
$("#leftLeg").show();
console.log("Left leg revealed.");
}
if (missCount > 6){
$("#rightLeg").show();
console.log("Right leg revealed.");
alert("Sorry... you lost! The word was " + randomWord + ". Better luck next time!~");
gameOver = true;
}
}
